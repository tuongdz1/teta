\begin{groupplot}[
	  group style={
	    group size=3 by 1,
	    horizontal sep=1.2cm,
	    ylabels at=edge left,
	    yticklabels at=edge left
	  },
	  width=0.32\linewidth,
	  height=0.42\linewidth,
	  xmin=0, xmax=6.3,
	  ymin=-1.5, ymax=1.5,
	  xlabel={$x$},
	  ylabel={$y$},
	  domain=0:6.3,
	  samples=180
]
  % Shared data points
  \pgfplotsset{data points/.style={only marks, mark=*, mark size=3.0pt, color=WarmOrange, mark options={draw=WarmInk, fill=WarmOrange!85}}}
  \def\datapoints{(0.5,0.53) (1.2,0.85) (1.8,0.99) (2.4,0.58) (3.1,0.09) (3.6,-0.49) (4.2,-0.84) (4.9,-1.00) (5.4,-0.70) (6.0,-0.26)}

  \nextgroupplot[
    warmaxis,
    title={(a) Underfit (too simple)}
  ]
    \addplot [warmcurve, color=WarmInk, dashed] {sin(deg(x))};
    \addplot [line width=2.2pt, color=WarmRose] {0.35*(x-3.1)};
    \addplot[data points] coordinates {\datapoints};

  \nextgroupplot[
    warmaxis,
    title={(b) Better fit}
  ]
    \addplot [warmcurve, color=WarmInk, dashed] {sin(deg(x))};
    \addplot [line width=2.2pt, color=WarmTeal] {sin(deg(x)) + 0.08*cos(deg(2*x))};
    \addplot[data points] coordinates {\datapoints};

  \nextgroupplot[
    warmaxis,
    title={(c) Overfit (wiggly)}
  ]
    \addplot [warmcurve, color=WarmInk, dashed] {sin(deg(x))};
    \addplot [line width=2.2pt, color=WarmOrange] {sin(deg(x)) + 0.45*sin(deg(8*x))};
    \addplot[data points] coordinates {\datapoints};
